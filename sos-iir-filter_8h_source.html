58int sos_filter_f32(float *input, float *output, int len, const SOS_Coefficients &coeffs, SOS_Delay_State &w) {
59 // Assumes a0 and b0 coefficients are one (1.0)
60 float f0 = coeffs.b1;
61 float f1 = coeffs.b2;
62 float f2 = coeffs.a1;
63 float f3 = coeffs.a2;
64 float f4 = w.w0;
65 float f5 = w.w1;
66 for (; len > 0; len--) {
67 float f6 = *input++;
68 f6 += f2 * f4; // coeffs.a1 * w0
69 f6 += f3 * f5; // coeffs.a2 * w1
70 float f7 = f6; // b0 assumed 1.0
71 f7 += f0 * f4; // coeffs.b1 * w0
72 f7 += f1 * f5; // coeffs.b2 * w1 -> result
73 *output++ = f7;
74 f5 = f4; // w1 = w0
75 f4 = f6; // w0 = f6
76 }
77 w.w0 = f4;
78 w.w1 = f5;
79 return 0;
80};

92float sos_filter_sum_sqr_f32(float *input, float *output, int len, const SOS_Coefficients &coeffs, SOS_Delay_State &w, float gain) {
93 // Assumes a0 and b0 coefficients are one (1.0)
94 float f0 = coeffs.b1;
95 float f1 = coeffs.b2;
96 float f2 = coeffs.a1;
97 float f3 = coeffs.a2;
98 float f4 = w.w0;
99 float f5 = w.w1;
100 float f6 = gain;
101 float sum_sqr = 0;
102 for (int i = len; i > 0; i--) {
103 float f7 = *input++;
104 f7 += f2 * f4; // coeffs.a1 * w0
105 f7 += f3 * f5; // coeffs.a2 * w1;
106 float f8 = f7; // b0 assumed 1.0
107 f8 += f0 * f4; // coeffs.b1 * w0;
108 f8 += f1 * f5; // coeffs.b2 * w1;
109 float f9 = f8 * f6; // f8 * gain -> result
110 *output++ = f9;
111 f5 = f4; // w1 = w0
112 f4 = f7; // w0 = f7;
113 sum_sqr += f9 * f9; // sum_sqr += f9 * f9;
114 }
115 w.w0 = f4;
116 w.w1 = f5;
117 return sum_sqr;
118};

140 SOS_IIR_Filter(size_t num_sos, const float gain, const SOS_Coefficients _sos[] = NULL)
141 : num_sos(num_sos), gain(gain) {
142 if (num_sos > 0) {
143 sos = new SOS_Coefficients[num_sos];
144 if ((sos != NULL) && (_sos != NULL)) memcpy(sos, _sos, num_sos * sizeof(SOS_Coefficients));

165 inline float filter(float *input, float *output, size_t len) {
166 if ((num_sos < 1) || (sos == NULL) || (w == NULL)) return 0;
167 float *source = input;
168 // Apply all but last Second-Order-Section
169 for (int i = 0; i < (num_sos - 1); i++) {
170 sos_filter_f32(source, output, len, sos[i], w[i]);
171 source = output;
172 }
173 // Apply last SOS with gain and return the sum of squares of all samples
174 return sos_filter_sum_sqr_f32(source, output, len, sos[num_sos - 1], w[num_sos - 1], gain);
175 }
176
180 ~SOS_IIR_Filter() {
181 if (w != NULL) delete[] w;
182 if (sos != NULL) delete[] sos;
183 }

205 inline float filter(float *input, float *output, size_t len) {
206 float sum_sqr = 0;
207 float s;
208 for (int i = 0; i < len; i++) {
209 s = input[i];
210 sum_sqr += s * s;
211 }
212 if (input != output) {
213 for (int i = 0; i < len; i++) output[i] = input[i];
214 }
215 return sum_sqr;
216 };
