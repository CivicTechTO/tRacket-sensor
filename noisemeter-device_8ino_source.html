98void setup() {
99 pinMode(PIN_LED1, OUTPUT);
100 digitalWrite(PIN_LED1, LOW);
101
102 // Grounding this pin (e.g. with a button) will force access open to start.
103 // Useful as a "reset" button to overwrite currently saved credentials.
104 pinMode(PIN_BUTTON, INPUT_PULLUP);
105
106 SERIAL.begin(115200);
107 delay(2000);
108 SERIAL.println();
109 SERIAL.print("Noisemeter ");
110 SERIAL.print(NOISEMETER_VERSION);
111 SERIAL.print(' ');
112 SERIAL.println(buildDeviceId());
113 SERIAL.println("Initializing...");
114
115 Creds.begin(buildDeviceId());
116#ifdef STORAGE_SHOW_CREDENTIALS
117 SERIAL.println(Creds);
118#endif
119
120 SPL.initMicrophone();
121 packets.emplace_front();
122
123#ifndef UPLOAD_DISABLED
124 bool isAPNeeded = false;
125
126 if (!digitalRead(PIN_BUTTON) || !Creds.valid()) {
127 SERIAL.print("Erasing stored credentials...");
128 Creds.clear();
129 SERIAL.println(" done.");
130
131 isAPNeeded = true;
132 } else if (Creds.get(Storage::Entry::Token)[0] == '\0') {
133 isAPNeeded = true;
134 } else if (tryWifiConnection(WIFI_STA) < 0) {
135 isAPNeeded = true;
136 } else if (Timestamp::synchronize() < 0) {
137 isAPNeeded = true;
138 }
139
140 // Run the access point if it is requested or if there are no valid credentials.
141 if (isAPNeeded) {
142 AccessPoint ap (saveNetworkCreds);
143 Blinker bl (500);
144
145 ap.run();
146
147 // Access point timed out: power off.
148 SERIAL.println("No connections. Power off.");
149 delay(500);
150 esp_sleep_disable_wakeup_source(ESP_SLEEP_WAKEUP_ALL);
151 esp_deep_sleep_start();
152 }
153
154 Timestamp now;
155 lastUpload = now;
156 lastOTACheck = now;
157 firstSend = true;
158
159 SERIAL.println("Connected to the WiFi network.");
160 SERIAL.print("Local ESP32 IP: ");
161 SERIAL.println(WiFi.localIP());
162 SERIAL.print("Current time: ");
163 SERIAL.println(now);
164#endif // !UPLOAD_DISABLED
165
166 digitalWrite(PIN_LED1, HIGH);
167}

173void loop() {
174 if (auto db = SPL.readMicrophoneData(); db) {
175 packets.front().add(*db);
176 printReadingToConsole(*db);
177 }
178
179#ifndef UPLOAD_DISABLED
180 const auto now = Timestamp();
181
182 if (lastUpload.secondsBetween(now) >= UPLOAD_INTERVAL_SEC) {
183 packets.front().timestamp = now;
184
185 if (WiFi.status() != WL_CONNECTED) {
186 SERIAL.println("Attempting WiFi reconnect...");
187 WiFi.reconnect();
188 delay(5000);
189 }
190
191 if (WiFi.status() == WL_CONNECTED) {
192 API api (buildDeviceId(), Creds.get(Storage::Entry::Token));
193
194 if (firstSend) {
195 if (api.sendMeasurementWithDiagnostics(packets.back(), NOISEMETER_VERSION, lastUpload)) {
196 packets.pop_back();
197 firstSend = false;
198 }
199 } else {
200 std::optional<Blinker> bl;
201
202 // Only blink if there's multiple packets to send
203 if (++packets.cbegin() != packets.cend())
204 bl.emplace(200);
205
206 packets.remove_if([&api](const auto& pkt) {
207 return pkt.count <= 0 || api.sendMeasurement(pkt);
208 });
209 }
210
211#if defined(BOARD_ESP32_PCB)
212 // We have WiFi: also check for software updates
213 if (lastOTACheck.secondsBetween(now) >= OTA_INTERVAL_SEC) {
214 lastOTACheck = now;
215 SERIAL.println("Checking for updates...");
216
217 const auto ota = api.getLatestSoftware();
218 if (ota) {
219 if (ota->version.compareTo(NOISEMETER_VERSION) > 0) {
220 SERIAL.print(ota->version);
221 SERIAL.println(" available!");
222
223 if (downloadOTAUpdate(ota->url, api.rootCertificate())) {
224 SERIAL.println("Download success! Restarting...");
225 delay(1000);
226 ESP.restart();
227 } else {
228 SERIAL.println("Update download failed.");
229 }
230 } else {
231 SERIAL.println("No new updates.");
232 }
233 } else {
234 SERIAL.println("Failed to reach update server!");
235 }
236 }
237#endif // BOARD_ESP32_PCB
238 }
239
240 if (!packets.empty()) {
241 const auto count = std::distance(packets.cbegin(), packets.cend());
242 SERIAL.print(count);
243 SERIAL.println(" packets still need to be sent!");
244
245 if (count >= MAX_SAVED_PACKETS) {
246 SERIAL.println("Discarded a packet!");
247 packets.pop_back();
248 }
249 }
250
251 // Create new packet for next measurements
252 packets.emplace_front();
253 lastUpload = now;
254 }
255#endif // !UPLOAD_DISABLED
256}

258void printReadingToConsole(double reading) {
259 String output = "";
260 output += std::lround(reading);
261 output += "dB";
262
263 const auto currentCount = packets.front().count;
264 if (currentCount > 1) {
265 output += " [+" + String(currentCount - 1) + " more]";
266 }
267 SERIAL.println(output);
268}

270std::optional<const char *> saveNetworkCreds(String ssid, String psk, String email)
271{
272 // Confirm that the given credentials will fit in the allocated EEPROM space.
273 if (!ssid.isEmpty() && Creds.canStore(ssid) && Creds.canStore(psk)) {
274 Creds.set(Storage::Entry::SSID, ssid);
275 Creds.set(Storage::Entry::Passkey, psk);
276 Creds.set(Storage::Entry::Unused, {});
277 Creds.commit();
278
279 if (tryWifiConnection(WIFI_AP_STA, WIFI_NEW_CONNECT_TIMEOUT_SEC) == 0) {
280 if (Timestamp::synchronize() == 0) {
281 if (email.length() > 0) {
282 API api (buildDeviceId());
283
284 if (const auto reg = api.sendRegister(email); reg) {
285 SERIAL.println("Registered!");
286 Creds.set(Storage::Entry::Token, *reg);
287 Creds.commit();
288
289 return {};
290 } else {
291 return "The sensor was not able to register with the server.";
292 }
293 } else if (Creds.get(Storage::Entry::Token)[0] != '\0') {
294 return {};
295 } else {
296 return "An email is required to register the sensor with the server.";
297 }
298 } else {
299 return "The sensor connected to your WiFi, but failed to reach the internet.";
300 }
301 } else {
302 return "The sensor was not able to connect to your WiFi using the info entered.";
303 }
304 } else {
305 return "No network name given, or network name or password is too long.";
306 }
307}

309UUID buildDeviceId()
310{
311 std::array<uint8_t, 6> mac;
312 esp_efuse_read_field_blob(ESP_EFUSE_MAC_FACTORY, mac.data(), /* bits */ mac.size() * 8);
313 return UUID(mac[0] | (mac[1] << 8) | (mac[2] << 16), mac[3] | (mac[4] << 8) | (mac[5] << 16));
314}

316int tryWifiConnection(wifi_mode_t mode, int timeout)
317{
318 WiFi.mode(mode);
319 const auto stat = WiFi.begin(
320 Creds.get(Storage::Entry::SSID).c_str(),
321 Creds.get(Storage::Entry::Passkey).c_str());
322 if (stat == WL_CONNECT_FAILED)
323 return -1;
324
325 // wait for WiFi connection
326 SERIAL.print("Waiting for WiFi to connect...");
327 const auto start = millis();
328 bool connected;
329
330 timeout = SEC_TO_MS(timeout);
331 do {
332 connected = WiFi.status() == WL_CONNECTED;
333 SERIAL.print(".");
334 delay(500);
335 } while (!connected && millis() - start < timeout);
336
337 return connected ? 0 : -1;
338}
