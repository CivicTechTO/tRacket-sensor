133 void setup() {
134  pinMode(PIN_LED1, OUTPUT);
135  digitalWrite(PIN_LED1, LOW);
136 
137  // Grounding this pin (e.g. with a button) will force access open to start.
138  // Useful as a "reset" button to overwrite currently saved credentials.
139  pinMode(PIN_BUTTON, INPUT_PULLUP);
140 
141  // If needed, now you can actually lower the CPU frquency,
142  // i.e. if you want to (slightly) reduce ESP32 power consumption
143  // setCpuFrequencyMhz(80); // It should run as low as 80MHz
144 
145  SERIAL.begin(115200);
146  delay(2000);
147  SERIAL.println();
148  SERIAL.print("Noisemeter ");
149  SERIAL.print(NOISEMETER_VERSION);
150  SERIAL.print(' ');
151  SERIAL.println(buildDeviceId());
152  SERIAL.println("Initializing...");
153 
154  Creds.begin();
155 
156  SPL.initMicrophone();
157  packets.emplace_front();
158 
159 #ifndef UPLOAD_DISABLED
160  bool isAPNeeded = false;
161 
162  if (!digitalRead(PIN_BUTTON) || !Creds.valid()) {
163  SERIAL.print("Erasing stored credentials...");
164  Creds.clear();
165  SERIAL.println(" done.");
166 
167  isAPNeeded = true;
168  } else if (tryWifiConnection() < 0 || Timestamp::synchronize() < 0) {
169  isAPNeeded = true;
170  }
171 
172  // Run the access point if it is requested or if there are no valid credentials.
173  if (isAPNeeded) {
174  AccessPoint ap (saveNetworkCreds);
175  Blinker bl (500);
176 
177  ap.run(); // does not return
178  }
179 
180  Timestamp now;
181  lastUpload = now;
182  lastOTACheck = now;
183 
184  SERIAL.println("Connected to the WiFi network.");
185  SERIAL.print("Local ESP32 IP: ");
186  SERIAL.println(WiFi.localIP());
187  SERIAL.print("Current time: ");
188  SERIAL.println(now);
189 #endif // !UPLOAD_DISABLED
190 
191  digitalWrite(PIN_LED1, HIGH);
192 }

198 void loop() {
199  if (auto db = SPL.readMicrophoneData(); db) {
200  packets.front().add(*db);
201  printReadingToConsole(*db);
202  }
203 
204 #ifndef UPLOAD_DISABLED
205  const auto now = Timestamp();
206 
207  if (lastUpload.secondsBetween(now) >= UPLOAD_INTERVAL_SEC) {
208  packets.front().timestamp = now;
209 
210  if (WiFi.status() != WL_CONNECTED) {
211  SERIAL.println("Attempting WiFi reconnect...");
212  WiFi.reconnect();
213  delay(5000);
214  }
215 
216  if (WiFi.status() == WL_CONNECTED) {
217  {
218  std::optional<Blinker> bl;
219 
220  // Only blink if there's multiple packets to send
221  if (++packets.cbegin() != packets.cend())
222  bl.emplace(300);
223 
224  packets.remove_if([](const auto& pkt) {
225  if (pkt.count > 0) {
226  const auto payload = createJSONPayload(pkt);
227  return uploadData(payload) == 0;
228  } else {
229  return true; // Discard empty packets
230  }
231  });
232  }
233 
234 #if defined(BOARD_ESP32_PCB)
235  // We have WiFi: also check for software updates
236  if (lastOTACheck.secondsBetween(now) >= OTA_INTERVAL_SEC) {
237  lastOTACheck = now;
238  SERIAL.println("Checking for updates...");
239 
240  OTAUpdate ota (cert_ISRG_Root_X1);
241  if (ota.available()) {
242  SERIAL.print(ota.version);
243  SERIAL.println(" available!");
244 
245  if (ota.download()) {
246  SERIAL.println("Download success! Restarting...");
247  delay(1000);
248  ESP.restart();
249  } else {
250  SERIAL.println("Update download failed.");
251  }
252  } else {
253  SERIAL.println("No update available.");
254  }
255  }
256 #endif // BOARD_ESP32_PCB
257  }
258 
259  if (!packets.empty()) {
260  const auto count = std::distance(packets.cbegin(), packets.cend());
261  SERIAL.print(count);
262  SERIAL.println(" packets still need to be sent!");
263 
264  if (count >= MAX_SAVED_PACKETS) {
265  SERIAL.println("Discarded a packet!");
266  packets.pop_back();
267  }
268  }
269 
270  // Create new packet for next measurements
271  packets.emplace_front();
272  lastUpload = now;
273  }
274 #endif // !UPLOAD_DISABLED
275 }

299 void saveNetworkCreds(WebServer& httpServer) {
300  // Confirm that the form was actually submitted.
301  if (httpServer.hasArg("ssid") && httpServer.hasArg("psk")) {
302  const auto id = String(buildDeviceId());
303  const auto ssid = Secret::encrypt(id, httpServer.arg("ssid"));
304  const auto psk = Secret::encrypt(id, httpServer.arg("psk"));
305 
306  // Confirm that the given credentials will fit in the allocated EEPROM space.
307  if (!ssid.isEmpty() && Creds.canStore(ssid) && Creds.canStore(psk)) {
308  Creds.set(Storage::Entry::SSID, ssid);
309  Creds.set(Storage::Entry::Passkey, psk);
310  Creds.commit();
311 
312  ESP.restart(); // Software reset.
313  }
314  }
315 
316  // TODO inform user that something went wrong...
317  SERIAL.println("Error: Invalid network credentials!");
318 }

320 UUID buildDeviceId()
321 {
322  std::array<uint8_t, 6> mac;
323  esp_efuse_read_field_blob(ESP_EFUSE_MAC_FACTORY, mac.data(), /* bits */ mac.size() * 8);
324  return UUID(mac[0] | (mac[1] << 8) | (mac[2] << 16), mac[3] | (mac[4] << 8) | (mac[5] << 16));
325 }

343  if (stat == WL_CONNECT_FAILED)
344  return -1;
345 
346  // wait for WiFi connection
347  SERIAL.print("Waiting for WiFi to connect...");
348  const auto start = millis();
349  bool connected;
350 
351  do {
352  connected = WiFi.status() == WL_CONNECTED;
353  SERIAL.print(".");
354  delay(500);
355  } while (!connected && millis() - start < WIFI_CONNECT_TIMEOUT_MS);
356 
357  return connected ? 0 : -1;
358 }

360 String createJSONPayload(const DataPacket& dp)
361 {
362 #ifdef BUILD_PLATFORMIO
363  JsonDocument doc;
364 #else
365  DynamicJsonDocument doc (2048);
366 #endif
367 
368  doc["parent"] = "/Bases/nm1";
369  doc["data"]["type"] = "comand";
370  doc["data"]["version"] = "1.0";
371  doc["data"]["contents"][0]["Type"] = "Noise";
372  doc["data"]["contents"][0]["Min"] = std::lround(dp.minimum);
373  doc["data"]["contents"][0]["Max"] = std::lround(dp.maximum);
374  doc["data"]["contents"][0]["Mean"] = std::lround(dp.average);
375  doc["data"]["contents"][0]["DeviceID"] = String(buildDeviceId());
376  doc["data"]["contents"][0]["Timestamp"] = String(dp.timestamp);
377 
378  // Serialize JSON document
379  String json;
380  serializeJson(doc, json);
381  return json;
382 }

385 int uploadData(String json)
386 {
387  WiFiClientSecure client;
388  HTTPClient https;
389 
390  client.setCACert(cert_ISRG_Root_X1);
391 
392  SERIAL.print("[HTTPS] begin...\n");
393  if (https.begin(client, "https://noisemeter.webcomand.com/ws/put")) {
394  SERIAL.print("[HTTPS] POST...\n");
395 
396  // start connection and send HTTP header
397  https.addHeader("Authorization", String("Token ") + API_TOKEN);
398  https.addHeader("Content-Type", "application/json");
399  https.addHeader("Content-Length", String(json.length()));
400  https.addHeader("User-Agent", "ESP32");
401 
402  int httpCode = https.POST(json);
403 
404  // httpCode will be negative on error
405  if (httpCode > 0) {
406  // HTTP header has been send and Server response header has been handled
407  SERIAL.printf("[HTTPS] POST... code: %d\n", httpCode);
408 
409  // file found at server
410  if (httpCode == HTTP_CODE_OK || httpCode == HTTP_CODE_MOVED_PERMANENTLY) {
411  String payload = https.getString();
412  SERIAL.println(payload);
413  } else {
414  SERIAL.printf("[HTTPS] POST... failed, error: %s\n", https.errorToString(httpCode).c_str());
415  return -1;
416  }
417  } else {
418  SERIAL.printf("[HTTPS] POST... failed, error: %s\n", https.errorToString(httpCode).c_str());
419  return -1;
420  }
421 
422  https.end();
423  } else {
424  SERIAL.printf("[HTTPS] Unable to connect\n");
425  return -1;
426  }
427
